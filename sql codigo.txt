CREATE TABLE estudiantes (
id INT IDENTITY(1,1),
documento VARCHAR(12),
nombre VARCHAR(50),
ciudad VARCHAR(30)
);

ALTER TABLE estudiantes 
ADD CONSTRAINT PK_EST_ID PRIMARY KEY(id);

CREATE TABLE carreras (
idcarrera INT IDENTITY(1,1) PRIMARY KEY,
nombre VARCHAR(50),
area VARCHAR(30)
);

CREATE TABLE universidades (
iduniversidad INT IDENTITY(1,1) PRIMARY KEY,
nombre VARCHAR(80),
ciudad VARCHAR(30)
);

CREATE TABLE interes (
idestudiante INT,
idcarrera INT,
PRIMARY KEY (idestudiante, idcarrera),
FOREIGN KEY (idestudiante) REFERENCES estudiantes(id),
FOREIGN KEY (idcarrera) REFERENCES carreras(idcarrera)
);

--cantidad estudiantes por ciudad 

SELECT ciudad, COUNT(*) AS total_estudiantes
FROM estudiantes
GROUP BY ciudad;

--cantidad estudiates por carrera 

SELECT c.nombre, COUNT(*) AS total
FROM interes i
INNER JOIN carreras c ON i.idcarrera = c.idcarrera
GROUP BY c.nombre;

--universidades con mas de 1 carrera 

SELECT u.nombre, COUNT(*) AS total
FROM universidad_carrera uc
INNER JOIN universidades u ON uc.iduniversidad = u.iduniversidad
GROUP BY u.nombre
HAVING COUNT(*) > 1;

--estudiante mas carrera 

SELECT e.nombre, c.nombre AS carrera
FROM interes i
INNER JOIN estudiantes e ON i.idestudiante = e.id
INNER JOIN carreras c ON i.idcarrera = c.idcarrera;

--intereses registrados hoy 

SELECT COUNT(*) AS total_hoy
FROM interes
WHERE fecha_interes = CAST(GETDATE() AS DATE);
